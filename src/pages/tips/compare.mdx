import { ForexStateKey, defaultForexState } from '@s/forex';

export const TuitionBarChart = dynamic(() =>
  import('@cnt/tips/compare/TuitionBarChart').then(
    ({ TuitionBarChart }) => TuitionBarChart,
  ),
);

export const getStaticProps = async(context) => {
  const atomStore = new AtomStore();
  const forex = process.env.ALPHA_VANTAGE_API_KEY
    ? await getForex('USD', 'JPY')
    : defaultForexState;
  atomStore.setAtom(ForexStateKey, forex);
  return atomStore.with({
    revalidate: 60 * 12
  });
}
